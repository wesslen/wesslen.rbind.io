library(urltools)

urls <- filter(t, !is.na(urls_expanded_url)) %>% 
    filter(created_at > "2018-01-01 00:00:00") %>%
  unnest(urls_expanded_url) %>% 
  select(urls_expanded_url, is_retweet, status_id, screen_name, created_at) %>%
  mutate(domains = domain(urls_expanded_url))

parsed <- url_parse(urls$urls_expanded_url)

urls %>%
  bind_cols(parsed) %>%
  filter(domains %in% c("youtu.be","www.youtube.com")) %>%
  select(urls_expanded_url)
